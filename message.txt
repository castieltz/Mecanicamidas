		function limpar() {
			var tunagens = document.getElementsByName('tunagem');
			for (var i = 0; i < tunagens.length; i++) {
				tunagens[i].checked = false;
			}
			$("#copiado")[0].style.display = 'none'

			var valor = document.getElementById("valor");
			valor.value = 0;
			document.getElementById("copy-text").value = "";

		}

			$("#copiado")[0].style.display = 'none';

		}

		function calcular() {
			var Valor = 0;
			var tunagens = document.getElementsByName('tunagem');
			for (var i = 0; i < tunagens.length; i++) {
				if (tunagens[i].checked) {
					var valores_tunganes = tunagens[i].value.split("|");
					Valor += parseInt(valores_tunganes[0]);

				}
				$("#valorTotal").val(Valor);
				var valor_txt = document.getElementById("valor");
				valor_txt.value = "R$" + Valor.toLocaleString('pt-BR', { minimumFractionDigits: 2 });;
			}
		}
		function descontoParceria(){
			if ( $("#valorTotal").val() >= 350000){
				var desconto= $("#valorTotal").val() * 0.15
				var Valor = $("#valorTotal").val() - desconto
				var valor_txt = document.getElementById("valor");
				valor_txt.value = "R$" + Valor.toLocaleString('pt-BR', { minimumFractionDigits: 2 });;
			}
		}
		
		function descontoDezPorcento(){
			var tunagem = document.getElementById("Tunagem Full /S Desconto").checked;

				if(tunagem === true ) {
					var Valor = $("#valorTotal").val() - desconto - 5000
				} else
				var Valor = $("#valorTotal").val() - desconto
				
				var valor_txt = document.getElementById("valor");
				valor_txt.value = "R$" + Valor.toLocaleString('pt-BR', { minimumFractionDigits: 2 });;
			}
